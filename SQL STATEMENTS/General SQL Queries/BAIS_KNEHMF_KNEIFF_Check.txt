select  A.KNEHMF_KDNRH,A.KNEHMF_NAME1,A.KNEHMF_WSGSI,B.KNEIFF_WSGSI,'Result_WSGSI'=Case when A.KNEHMF_WSGSI=B.KNEIFF_WSGSI then 'OK' else 'NO MATCH' END
,A.KNEHMF_BRNCH,B.KNEIFF_BRNCH,'Result_BRNCH'=Case when A.KNEHMF_BRNCH=B.KNEIFF_BRNCH then 'OK' else 'NO MATCH' END
,A.KNEHMF_WSBIS,B.KNEIFF_WSBIS,'Result_WSBIS'=Case when A.KNEHMF_WSBIS=B.KNEIFF_WSBIS then 'OK' else 'NO MATCH' END 
from [CCB-BAIS-PROD].[baisjdb].dbo.KNEHMF A
INNER JOIN [BAIS_KNEIFF] B on A.KNEHMF_KDNRH=B.KNEIFF_KDNRH where B.[KNEIFF_DXIFD]='20170823' and LEN(A.KNEHMF_KDNRH)>5
